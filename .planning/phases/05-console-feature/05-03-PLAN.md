## AgentResult Type (from 05-01)

```typescript
interface AgentResult {
  traceId?: string
  startedAt?: string
  completedAt?: string
  durationMs?: number
  success?: boolean
  needsModelInput?: boolean
  response?: string
  plan?: string[]
  toolCalls?: AgentToolCall[]
  metrics?: AgentMetrics
  clarification?: { question?: string; missingFields?: string[] }
  artifacts?: Array<{ format: string; path: string }>
  report?: { summary?: string; markdown?: string }
}

interface AgentMetrics {
  toolCount?: number
  failedToolCount?: number
  totalToolDurationMs?: number
  averageToolDurationMs?: number
  maxToolDurationMs?: number
}

interface AgentToolCall {
  tool: string
  status: 'success' | 'error'
  durationMs?: number
  errorCode?: string
  error?: string
}
```
